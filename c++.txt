#include<iostream>
using namespace std;

int main() {
    int n;
    cin >> n;

    int i = 1;
    while (i<=n){
        //print space
        int space = n - i;
        while (space){
            cout<<" ";
            space = space - 1;

        }



        //print 1st triangle
        int j = 1;
        while (j<=i){
            cout<<j;
            j = j + 1;
        }


        //print 2nd triangle
        int start = i -1;
        while (start){
            cout<<start;
            start = start - 1;

        }
        cout<<endl;
        i = i + 1;
    }
    
}

// sum of digits
#include<iostream>
using namespace std;

int main() {
    int n;
    cin>>n;
    int sum = 0;
    while(n!=0){
        int digits = n % 10;
        sum = sum + digits;
        n = n / 10;
    }
    cout<<sum;
    
}

int main() {
    int n ;
    cin >> n;
    int ans = 0;
