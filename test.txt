#selective sorting

def sort(nums):
    
    for i in range(len(nums)):
        minval = i 
        for j in range(i,len(nums)):
            if nums[j]<nums[minval]:
                minval = j

        temp = nums[i]
        nums[i]= nums[minval]
        nums[minval] = temp


nums = [5,3,8,6,7,2]
sort(nums)

print(nums)
 

#bubble sorting

def sort(nums):
    for i in range(len(nums)- 1 , 0 , -1):
        for j in range(i):
            if nums[j]> nums[j+1]:
                temp = nums[j]
                nums[j]= nums[j+1]
                nums[j+1]= temp


nums = [5,2,6,8,9,4]
sort(nums)
print(nums)

def add(a,b):

    print(result)

def mul(a,b):
    result = a * b 
    print(result)

def div(a,b):
    result = a / b 
    print(result)
